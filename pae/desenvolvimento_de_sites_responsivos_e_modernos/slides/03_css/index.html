Cascata e Herança
            
              body {
                background: red;
              }

              p {
                color: blue;
              }

              a {
                font-size: 18px;
              }

              ul li a {
                font-size: 12px;
              }
Seletores
            
              body p {
                color: red;
              }

              body p.texto {
                color: black;
              }

              body p#texto {
                color: black;
              }

              div[data-teste="green"] {
                background-color: green;
              }

              body > h4 {
                font-size: 40px;
              }
              
              ul li a + span {
                font-size: 30px;
              }

              a:hover {
                color: red;
              }
